  codigo, filhos : inteiro;
  horas, valorHora, salario : real;
Início
  Imprimir("Informe o seu código: ");
  Ler(codigo);
  Imprimir("Informe a quantidade de horas trabalhadas no mês: ");
  Ler(horas);
  Imprimir("Informe o valor da sua hora: ");
  Ler(valorHora);
  Imprimir("Informe a quantidade de filhos menor que 14 anos: ");
  Ler(filhos);
  salario <- horas * valorHora + filhos * 15;
  Imprimir("O funcionário de código ", codigo, 
     "recerá o salário de R$", salario);
Fim.
