    Imprimir("Informe um número: ");
    Ler(numero);

    Se (i = 1) Então 
      Início
        maior := numero;
        menor := numero;
      Fim;
    Senão
      Se (numero > maior) Então
        maior := numero;
      Senão
        Se (numero < menor) Então
          menor := numero;
        FimSe;
      FimSe;
    FimSe;
  FimPara;
